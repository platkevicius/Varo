CREATE TABLE `User` (
	`UUID` VARCHAR(255) NOT NULL,
	`alive` BOOLEAN NOT NULL,
	`lastLogging` TIME NOT NULL,
	`online` BOOLEAN NOT NULL AUTO_INCREMENT,
	`x` BINARY NOT NULL,
	`y` BINARY NOT NULL,
	`z` BINARY NOT NULL,
	PRIMARY KEY (`UUID`)
);

CREATE TABLE `UserKills` (
	`id` BINARY NOT NULL AUTO_INCREMENT,
	`killer` VARCHAR(255) NOT NULL,
	`killed` VARCHAR(255) NOT NULL,
	PRIMARY KEY (`id`)
);

CREATE TABLE `LootBox` (
	`id` BINARY NOT NULL AUTO_INCREMENT,
	`x` BINARY NOT NULL,
	`y` BINARY NOT NULL,
	`z` BINARY NOT NULL,
	`opened` BOOLEAN NOT NULL,
	PRIMARY KEY (`id`)
);

CREATE TABLE `BorderCordinates` (
	`id` BINARY NOT NULL AUTO_INCREMENT,
	`x` BINARY NOT NULL,
	`y` BINARY NOT NULL,
	`z` BINARY NOT NULL,
	`radius` BINARY NOT NULL,
	PRIMARY KEY (`id`)
);

ALTER TABLE `UserKills` ADD CONSTRAINT `UserKills_fk0` FOREIGN KEY (`killer`) REFERENCES `User`(`UUID`);

ALTER TABLE `UserKills` ADD CONSTRAINT `UserKills_fk1` FOREIGN KEY (`killed`) REFERENCES `User`(`UUID`);
